\begin{table}[H]
\centering
\caption{Описание членов базового класса FormulaTeam (Абстрактный)}
\begin{tabular}{|p{4.5cm}|p{3.5cm}|p{2cm}|p{5cm}|}
\hline
\textbf{Название} & \textbf{Тип} & \textbf{Доступ} & \textbf{Описание} \\ \hline
\multicolumn{4}{|c|}{\textbf{Свойства}} \\ \hline
Id & int & public & Уникальный идентификатор \\ \hline
TeamName & string & public & Название команды \\ \hline
PrincipalName & string & public & Имя руководителя команды \\ \hline
Headquarters & string & public & Штаб-квартира команды \\ \hline
TeamColor & string? & public & Цвет команды \\ \hline
ChampionshipPoints & int & public & Очки чемпионата \\ \hline
RaceWins, Podiums & int & public & Победы и подиумы \\ \hline
Drivers & {\footnotesize List<FormulaDriver>} & public & Список гонщиков \\ \hline
\multicolumn{4}{|c|}{\textbf{Методы и Конструкторы}} \\ \hline
FormulaTeam(...) & - & protected & Конструкторы \\ \hline
GetSeriesName() & string & public & Абстрактный метод \\ \hline
CalculatePoints(pos: int) & int & public & Абстрактный метод \\ \hline
AddPoints(pts: int) & void & public & Добавление очков \\ \hline
AddWin(), AddPodium() & void & public & Изменение статистики \\ \hline
GetWinRate(...) & double & public & Расчет процента побед \\ \hline
GetSummary() & string & public & Сводка по команде (virtual) \\ \hline
operator >, operator < & bool & public & Сравнение команд по очкам \\ \hline
\end{tabular}
\end{table}
