\section{Цель работы}
Получить практические навыки проектирования последовательностных цифровых устройств. Синтезировать и исследовать функциональную схему 4-разрядного параллельного регистра памяти (памяти типа RG) на базе D-триггеров с общими шинами синхронизации и сброса.

\section{Условно Графическое Обозначение регистра}
Параллельный регистр предназначен для одновременного приема, хранения и выдачи 4-разрядного слова данных. Устройство содержит информационные входы $D_3 \dots D_0$, синхронизирующий вход записи $C$, асинхронный вход сброса в ноль $R$ и выходы $Q_3 \dots Q_0$.

\section{Логические уравнения функций возбуждения выходов}
Полная каноническая форма (КФ) переходов для каждого разряда регистра строится на основании таблицы истинности для наборов, где соответствующий выход принимает значение логической единицы. Логические функции выходов выражаются через конъюнкции управляющих сигналов $C$, $R$ и всех информационных разрядов данных:

\begin{align*}
  Q_3 = {} & (C \cdot R \cdot D_3 \cdot \overline{D}_2 \cdot \overline{D}_1 \cdot \overline{D}_0) \vee
             (C \cdot R \cdot D_3 \cdot \overline{D}_2 \cdot \overline{D}_1 \cdot D_0) \vee \\
           & (C \cdot R \cdot D_3 \cdot \overline{D}_2 \cdot D_1 \cdot \overline{D}_0) \vee
             (C \cdot R \cdot D_3 \cdot \overline{D}_2 \cdot D_1 \cdot D_0) \vee \\
           & (C \cdot R \cdot D_3 \cdot D_2 \cdot \overline{D}_1 \cdot \overline{D}_0) \vee
             (C \cdot R \cdot D_3 \cdot D_2 \cdot \overline{D}_1 \cdot D_0) \vee \\
           & (C \cdot R \cdot D_3 \cdot D_2 \cdot D_1 \cdot \overline{D}_0) \vee
             (C \cdot R \cdot D_3 \cdot D_2 \cdot D_1 \cdot D_0) \\[3mm]
  Q_2 = {} & (C \cdot R \cdot \overline{D}_3 \cdot D_2 \cdot \overline{D}_1 \cdot \overline{D}_0) \vee
             (C \cdot R \cdot \overline{D}_3 \cdot D_2 \cdot \overline{D}_1 \cdot D_0) \vee \\
           & (C \cdot R \cdot \overline{D}_3 \cdot D_2 \cdot D_1 \cdot \overline{D}_0) \vee
             (C \cdot R \cdot \overline{D}_3 \cdot D_2 \cdot D_1 \cdot D_0) \vee \\
           & (C \cdot R \cdot D_3 \cdot D_2 \cdot \overline{D}_1 \cdot \overline{D}_0) \vee
             (C \cdot R \cdot D_3 \cdot D_2 \cdot \overline{D}_1 \cdot D_0) \vee \\
           & (C \cdot R \cdot D_3 \cdot D_2 \cdot D_1 \cdot \overline{D}_0) \vee
             (C \cdot R \cdot D_3 \cdot D_2 \cdot D_1 \cdot D_0) \\[3mm]
  Q_1 = {} & (C \cdot R \cdot \overline{D}_3 \cdot \overline{D}_2 \cdot D_1 \cdot \overline{D}_0) \vee
             (C \cdot R \cdot \overline{D}_3 \cdot \overline{D}_2 \cdot D_1 \cdot D_0) \vee \\
           & (C \cdot R \cdot \overline{D}_3 \cdot D_2 \cdot D_1 \cdot \overline{D}_0) \vee
             (C \cdot R \cdot \overline{D}_3 \cdot D_2 \cdot D_1 \cdot D_0) \vee \\
           & (C \cdot R \cdot D_3 \cdot \overline{D}_2 \cdot D_1 \cdot \overline{D}_0) \vee
             (C \cdot R \cdot D_3 \cdot \overline{D}_2 \cdot D_1 \cdot D_0) \vee \\
           & (C \cdot R \cdot D_3 \cdot D_2 \cdot D_1 \cdot \overline{D}_0) \vee
             (C \cdot R \cdot D_3 \cdot D_2 \cdot D_1 \cdot D_0) \\[3mm]
  Q_0 = {} & (C \cdot R \cdot \overline{D}_3 \cdot \overline{D}_2 \cdot \overline{D}_1 \cdot D_0) \vee
             (C \cdot R \cdot \overline{D}_3 \cdot \overline{D}_2 \cdot D_1 \cdot D_0) \vee \\
           & (C \cdot R \cdot \overline{D}_3 \cdot D_2 \cdot \overline{D}_1 \cdot D_0) \vee
             (C \cdot R \cdot \overline{D}_3 \cdot D_2 \cdot D_1 \cdot D_0) \vee \\

             (C \cdot R \cdot D_3 \cdot \overline{D}_2 \cdot D_1 \cdot D_0) \vee \\
           & (C \cdot R \cdot D_3 \cdot D_2 \cdot \overline{D}_1 \cdot D_0) \vee
             (C \cdot R \cdot D_3 \cdot D_2 \cdot D_1 \cdot D_0)

\section{Минимизация логических функций}
Поскольку разряды параллельного регистра функционируют независимо друг от друга, влияние комбинаций соседних информационных линий взаимно компенсируется (склеивается) при минимизации. Выполним последовательное пошаговое группирование и вынесение общих множителей для каждого уравнения:

\begin{align*}
  Q_3 &= C \cdot R \cdot D_3 \cdot (\overline{D}_2\overline{D}_1\overline{D}_0 \vee \overline{D}_2\overline{D}_1D_0 \vee \overline{D}_2D_1\overline{D}_0 \vee \overline{D}_2D_1D_0 \vee D_2\overline{D}_1\overline{D}_0 \vee D_2\overline{D}_1D_0 \vee D_2D_1\overline{D}_0 \vee D_2D_1D_0) = \\
      &= C \cdot R \cdot D_3 \cdot [\overline{D}_2\overline{D}_1(\overline{D}_0 \vee D_0) \vee \overline{D}_2D_1(\overline{D}_0 \vee D_0) \vee D_2\overline{D}_1(\overline{D}_0 \vee D_0) \vee D_2D_1(\overline{D}_0 \vee D_0)] = \\
      &= C \cdot R \cdot D_3 \cdot [\overline{D}_2\overline{D}_1 \vee \overline{D}_2D_1 \vee D_2\overline{D}_1 \vee D_2D_1] = \\
      &= C \cdot R \cdot D_3 \cdot [\overline{D}_2(\overline{D}_1 \vee D_1) \vee D_2(\overline{D}_1 \vee D_1)] = C \cdot R \cdot D_3 \cdot [\overline{D}_2 \vee D_2] = C \cdot D_3 \cdot R \\[3mm]
  Q_2 &= C \cdot R \cdot D_2 \cdot (\overline{D}_3\overline{D}_1\overline{D}_0 \vee \overline{D}_3\overline{D}_1D_0 \vee \overline{D}_3D_1\overline{D}_0 \vee \overline{D}_3D_1D_0 \vee D_3\overline{D}_1\overline{D}_0 \vee D_3\overline{D}_1D_0 \vee D_3D_1\overline{D}_0 \vee D_3D_1D_0) = \\
      &= C \cdot R \cdot D_2 \cdot [\overline{D}_3\overline{D}_1(\overline{D}_0 \vee D_0) \vee \overline{D}_3D_1(\overline{D}_0 \vee D_0) \vee D_3\overline{D}_1(\overline{D}_0 \vee D_0) \vee D_3D_1(\overline{D}_0 \vee D_0)] = \\
      &= C \cdot R \cdot D_2 \cdot [\overline{D}_3\overline{D}_1 \vee \overline{D}_3D_1 \vee D_3\overline{D}_1 \vee D_3D_1] = \\
      &= C \cdot R \cdot D_2 \cdot [\overline{D}_3(\overline{D}_1 \vee D_1) \vee D_3(\overline{D}_1 \vee D_1)] = C \cdot R \cdot D_2 \cdot [\overline{D}_3 \vee D_3] = C \cdot D_2 \cdot R \\[3mm]
  Q_1 &= C \cdot R \cdot D_1 \cdot (\overline{D}_3\overline{D}_2\overline{D}_0 \vee \overline{D}_3\overline{D}_2D_0 \vee \overline{D}_3D_2\overline{D}_0 \vee \overline{D}_3D_2D_0 \vee D_3\overline{D}_2\overline{D}_0 \vee D_3\overline{D}_2D_0 \vee D_3D_2\overline{D}_0 \vee D_3D_2D_0) = \\
      &= C \cdot R \cdot D_1 \cdot [\overline{D}_3\overline{D}_2(\overline{D}_0 \vee D_0) \vee \overline{D}_3D_2(\overline{D}_0 \vee D_0) \vee D_3\overline{D}_2(\overline{D}_0 \vee D_0) \vee D_3D_2(\overline{D}_0 \vee D_0)] = \\
      &= C \cdot R \cdot D_1 \cdot [\overline{D}_3\overline{D}_2 \vee \overline{D}_3D_2 \vee D_3\overline{D}_2 \vee D_3D_2] = \\
      &= C \cdot R \cdot D_1 \cdot [\overline{D}_3(\overline{D}_2 \vee D_2) \vee D_3(\overline{D}_2 \vee D_2)] = C \cdot R \cdot D_1 \cdot [\overline{D}_3 \vee D_3] = C \cdot D_1 \cdot R \\[3mm]
  Q_0 &= C \cdot R \cdot D_0 \cdot (\overline{D}_3\overline{D}_2\overline{D}_1 \vee \overline{D}_3\overline{D}_2D_1 \vee \overline{D}_3D_2\overline{D}_1 \vee \overline{D}_3D_2D_1 \vee D_3\overline{D}_2\overline{D}_1 \vee D_3\overline{D}_2D_1 \vee D_3D_2\overline{D}_1 \vee D_3D_2D_1) = \\
      &= C \cdot R \cdot D_0 \cdot [\overline{D}_3\overline{D}_2(\overline{D}_1 \vee D_1) \vee \overline{D}_3D_2(\overline{D}_1 \vee D_1) \vee D_3\overline{D}_2(\overline{D}_1 \vee D_1) \vee D_3D_2(\overline{D}_1 \vee D_1)] = \\
      &= C \cdot R \cdot D_0 \cdot [\overline{D}_3\overline{D}_2 \vee \overline{D}_3D_2 \vee D_3\overline{D}_2 \vee D_3D_2] = \\
      &= C \cdot R \cdot D_0 \cdot [\overline{D}_3(\overline{D}_2 \vee D_2) \vee D_3(\overline{D}_2 \vee D_2)] = C \cdot R \cdot D_0 \cdot [\overline{D}_3 \vee D_3] = C \cdot D_0 \cdot R 
\end{align*}

Таким образом, каждый разряд регистра реализует независимую функцию перезаписи данных по тактовому сигналу при отсутствии активного сигнала сброса.

\section{Функциональная схема параллельного регистра}
Синтезированная функциональная схема строится на базе четырёх синхронных D-триггеров, обладающих входами асинхронного сброса. Информационные входы $D_3 \dots D_0$ подключаются напрямую к соответствующим $D$-входам триггеров. Линии тактовой синхронизации $C$ и асинхронного сброса $R$ объединяются параллельно, формируя общие управляющие шины устройства.
